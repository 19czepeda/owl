109  LinearSpace2 orthogonal() const {
110  LinearSpace2 m = *this;
111 
112  // mirrored?
113  scalar_t mirror(OneTy());
114  if (m.det() < scalar_t(ZeroTy())) {
115  m.vx = -m.vx;
116  mirror = -mirror;
117  }
118 
119  // rotation
120  for (int i = 0; i < 99; i++) {
121  const LinearSpace2 m_next = 0.5 * (m + m.transposed().inverse());
122  const LinearSpace2 d = m_next - m;
123  m = m_next;
124  // norm^2 of difference small enough?
125  if (max(dot(d.vx, d.vx), dot(d.vy, d.vy)) < 1e-8)
126  break;
127  }
128 
129  // rotation * mirror_x
130  return LinearSpace2(mirror*m.vx, m.vy);
131  }

267  static inline __both__ LinearSpace3 rotate(const vector_t& _u, const scalar_t& r) {
268  vector_t u = normalize(_u);
269  scalar_t s = sin(r), c = cos(r);
270  return LinearSpace3(u.x*u.x+(1-u.x*u.x)*c, u.x*u.y*(1-c)-u.z*s, u.x*u.z*(1-c)+u.y*s,
271  u.x*u.y*(1-c)+u.z*s, u.y*u.y+(1-u.y*u.y)*c, u.y*u.z*(1-c)-u.x*s,
272  u.x*u.z*(1-c)-u.y*s, u.y*u.z*(1-c)+u.x*s, u.z*u.z+(1-u.z*u.z)*c);
273  }

276  static inline __both__ QuaternionT<scalar_t> rotation(const LinearSpace3 &a) {
277  scalar_t tr = a.vx.x+a.vy.y+a.vz.z+1;
278  vector_t diag(a.vx.x,a.vy.y,a.vz.z);
279  if (tr > 1) {
280  scalar_t s = owl::common::polymorphic::sqrt(tr) * 2;
281  return QuaternionT<scalar_t>(.25f * s,
282  (a.vz.y-a.vy.z)/s,
283  (a.vx.z-a.vz.x)/s,
284  (a.vy.x-a.vx.y)/s);
285  } else if (arg_min(diag) == 0) {
286  scalar_t s = owl::common::polymorphic::sqrt(1.f+diag.x-diag.y-diag.z)*2.f;
287  return QuaternionT<scalar_t>((a.vz.y-a.vy.z)/s,
288  .25f * s,
289  (a.vx.y-a.vy.x)/s,
290  (a.vx.z-a.vz.x)/s);
291  } else if (arg_min(diag) == 1) {
292  scalar_t s = owl::common::polymorphic::sqrt(1.f+diag.y-diag.x-diag.z)*2.f;
293  return QuaternionT<scalar_t>((a.vx.z-a.vz.x)/s,
294  (a.vx.y-a.vy.x)/s,
295  .25f * s,
296  (a.vy.z-a.vz.y)/s);
297  } else {
298  scalar_t s = owl::common::polymorphic::sqrt(1.f+diag.z-diag.x-diag.y)*2.f;
299  return QuaternionT<scalar_t>((a.vy.x-a.vx.y)/s,
300  (a.vx.z-a.vz.x)/s,
301  (a.vy.z-a.vz.y)/s,
302  .25f * s);
303  }
304  }

320  /* constructs a coordinate frame form a normalized normal */
321  template<typename T>
322  inline __both__ LinearSpace3<T> frame(const T &N)
323  {
324 // #ifdef __CUDA_ARCH__
325  const T dx0 = cross(T(OneTy(),ZeroTy(),ZeroTy()),N);
326  const T dx1 = cross(T(ZeroTy(),OneTy(),ZeroTy()),N);
327 // #else
328 // const T dx0 = cross(T(one,zero,zero),N);
329 // const T dx1 = cross(T(zero,one,zero),N);
330 // #endif
331  const T dx = normalize(select(dot(dx0,dx0) > dot(dx1,dx1),dx0,dx1));
332  const T dy = normalize(cross(N,dx));
333  return LinearSpace3<T>(dx,dy,N);
334  }
335 
336  /* constructs a coordinate frame from a normal and approximate x-direction */
337  template<typename T> inline __both__ LinearSpace3<T> frame(const T& N, const T& dxi)
338  {
339  if (abs(dot(dxi,N)) > 0.99f) return frame(N); // fallback in case N and dxi are very parallel
340  const T dx = normalize(cross(dxi,N));
341  const T dy = normalize(cross(N,dx));
342  return LinearSpace3<T>(dx,dy,N);
343  }
